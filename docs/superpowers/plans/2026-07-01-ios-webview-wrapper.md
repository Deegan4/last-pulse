# Last Pulse iOS WKWebView Wrapper Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the existing `index.html` canvas game as an installable iOS app (TestFlight-distributable) by wrapping it, unmodified in gameplay terms, in a thin native SwiftUI + WKWebView shell with haptics and a Game Center "Best Placement" leaderboard.

**Architecture:** A new `LastPulse.xcodeproj` SwiftUI iOS app target loads `index.html` from the app bundle into a `WKWebView`, fully offline. A `WKScriptMessageHandler` bridge receives one-way JS→Swift messages (`haptic`, `submitScore`) from two new call sites added to `index.html`. No native code drives gameplay; the web game remains the sole source of truth.

**Tech Stack:** Swift, SwiftUI, WebKit (`WKWebView`), GameKit (`GKLocalPlayer`, `GKLeaderboard`), Xcode 26 project format. No CocoaPods/SPM dependencies needed — everything used is a first-party Apple framework.

## Global Constraints

- Offline-first: `index.html` and all its bundled assets (`assets/meshy/*.glb`, `assets/meshy/loader.js`) must load via `loadFileURL`, not over network.
- The bridge is one-way (JS → Swift only). Native code never injects JS back into the page.
- All bridge calls from `index.html` use optional chaining (`window.webkit?.messageHandlers?.native?.postMessage(...)`) so the game still runs unmodified in a plain browser (`open index.html`, `python3 -m http.server`).
- Leaderboard identifier: `bestPlacement`, ascending sort (lower placement number = better rank). Only submitted for non-horde matches (`place` is not meaningful in horde mode — see `index.html:1170-1180`).
- No automated test suite is added, matching this repo's existing manual-validation convention (`scripts/parse-check.mjs` + headless render). Each task ends with a manual verification step instead of an automated test.
- Portrait-only orientation, matching the existing web game's portrait-first layout.

```swift
import SwiftUI
import WebKit

struct GameWebView: UIViewRepresentable {
    let onLoadFailure: () -> Void
    let messageHandler: WKScriptMessageHandler

    func makeUIView(context: Context) -> WKWebView {
        let config = WKWebViewConfiguration()
        config.userContentController.add(messageHandler, name: "native")

        let webView = WKWebView(frame: .zero, configuration: config)
        webView.scrollView.bounces = false
        webView.scrollView.pinchGestureRecognizer?.isEnabled = false
        webView.isOpaque = false
        webView.backgroundColor = UIColor(red: 0.04, green: 0.04, blue: 0.06, alpha: 1)
        webView.navigationDelegate = context.coordinator

        guard let url = Bundle.main.url(
            forResource: "index",
            withExtension: "html",
            subdirectory: "GameResources"
        ) else {
            onLoadFailure()
            return webView
        }
        webView.loadFileURL(url, allowingReadAccessTo: url.deletingLastPathComponent())
        return webView
    }

    func updateUIView(_ uiView: WKWebView, context: Context) {}

    func makeCoordinator() -> Coordinator {
        Coordinator(onLoadFailure: onLoadFailure)
    }

    final class Coordinator: NSObject, WKNavigationDelegate {
        let onLoadFailure: () -> Void
        init(onLoadFailure: @escaping () -> Void) { self.onLoadFailure = onLoadFailure }

        func webView(_ webView: WKWebView, didFail navigation: WKNavigation!, withError error: Error) {
            onLoadFailure()
        }
        func webView(_ webView: WKWebView, didFailProvisionalNavigation navigation: WKNavigation!, withError error: Error) {
            onLoadFailure()
        }
    }
}
```

```swift
import Foundation
import WebKit
import UIKit

final class NativeBridge: NSObject, WKScriptMessageHandler {
    private let impactGenerator = UIImpactFeedbackGenerator(style: .medium)
    private let notificationGenerator = UINotificationFeedbackGenerator()

    func userContentController(_ controller: WKUserContentController, didReceive message: WKScriptMessage) {
        guard let body = message.body as? [String: Any],
              let type = body["type"] as? String else { return }

        switch type {
        case "haptic":
            handleHaptic(style: body["style"] as? String)
        case "submitScore":
            handleSubmitScore(placement: body["placement"] as? Int)
        default:
            break
        }
    }

    private func handleHaptic(style: String?) {
        switch style {
        case "impact":
            impactGenerator.impactOccurred()
        case "success":
            notificationGenerator.notificationOccurred(.success)
        default:
            break
        }
    }

    private func handleSubmitScore(placement: Int?) {
        guard let placement else { return }
        GameCenterManager.shared.submitPlacement(placement)
    }
}
```

```swift
import Foundation
import GameKit

enum GameCenterManager {
    static let shared = GameCenterManagerImpl()
}

final class GameCenterManagerImpl {
    private static let bestPlacementLeaderboardID = "bestPlacement"

    func authenticate() {
        GKLocalPlayer.local.authenticateHandler = { viewController, error in
            guard let viewController else { return }
            guard let rootVC = UIApplication.shared.connectedScenes
                .compactMap({ ($0 as? UIWindowScene)?.keyWindow?.rootViewController })
                .first else { return }
            rootVC.present(viewController, animated: true)
        }
    }

    func submitPlacement(_ placement: Int) {
        guard GKLocalPlayer.local.isAuthenticated else { return }
        GKLeaderboard.submitScore(
            placement,
            context: 0,
            player: GKLocalPlayer.local,
            leaderboardIDs: [Self.bestPlacementLeaderboardID]
        ) { error in
            // Fire-and-forget per spec: no retry queue, no UI feedback on failure.
        }
    }
}
```
